#include using namespace std; int binarySearch(int arr[], int left, int right, int target) { while (left <= right) { int mid = left + (right - left) / 2; if (arr[mid] == target) { return mid; } else if (arr[mid] < target) { left = mid + 1; } else { right = mid - 1; } } return -1; } int main() { int arr[] = {7, 10, 15, 19, 23, 29}; int n = sizeof(arr) / sizeof(arr[0]); int target = 19; int result = binarySearch(arr, 0, n - 1, target); if (result != -1) { cout << "Element found at index " << result << endl; } else { cout << "Element not found in the array" << endl; } return 0; }

#include using namespace std; int linearSearch(int arr[], int n, int target) { for (int i = 0; i < n; i++) { if (arr[i] == target) { return i; } } return -1; } int main() { int arr[] = {10, 23, 19, 15, 7, 29}; int n = sizeof(arr) / sizeof(arr[0]); int target = 15; int result = linearSearch(arr, n, target); if (result != -1) { cout << "Element found at index " << result << endl; } else { cout << "Element not found in the array" << endl; } return 0; }
